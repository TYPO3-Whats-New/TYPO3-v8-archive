\begin{frame}[fragile]
	\frametitle{Veraltete/Entfernte Funktionen}
	\framesubtitle{Verschiedenes}

	\begin{itemize}

		\item Die folgenden Konfigurations-Optionen wurden entfernt (hier findet eine Autodection statt und es wird per Default \texttt{mbstring} gewählt):

			\begin{itemize}
				\item \texttt{\$TYPO3\_CONF\_VARS['SYS']['t3lib\_cs\_utils']}
				\item \texttt{\$TYPO3\_CONF\_VARS['SYS']['t3lib\_cs\_convMethod']}
			\end{itemize}

		\item Die veraltete TypoScript-Eigenschaft \texttt{page.includeJSlibs} wurde
			entfernt. Stattdessen muss man nun die TypoScript-Eigenschaft \texttt{page.includeJSLibs}
			(mit großem "L") verwenden

		\item Die TypoScript-Option \texttt{config.renderCharset} wurde entfernt

	\end{itemize}

\end{frame}
