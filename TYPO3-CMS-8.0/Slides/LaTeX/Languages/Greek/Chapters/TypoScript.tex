\begin{frame}[fragile]
	\frametitle{TSconfig \& TypoScript}
	\framesubtitle{\texttt{HTMLparser.stripEmptyTags.keepTags}}

	% decrease font size for code listing
	\lstset{basicstyle=\tiny\ttfamily}

	\begin{itemize}

		\item \greektext Έχει προστεθεί νέα επιλογή για τη διαμόρφωση \latintext \texttt{HTMLparser.stripEmptyTags},
			\greektext η οποία επιτρέπει τη διατήρηση διαμορφωμένων \latintext tags
		\item \greektext Πριν από αυτή την αλλαγή, μόνο μία λίστα από \latintext tags \greektext μπορούσε να
		παρασχεθεί, η οποία έπρεπε να αφαιρεθεί
		\item \greektext Το ακόλουθο παράδειγμα αφαιρεί όλα τα κενά \latintext tags \greektext\textbf{εκτός} των \latintext\texttt{tr}
		\greektext και \latintext \texttt{td}:

			\latintext\begin{lstlisting}
				HTMLparser.stripEmptyTags = 1
				HTMLparser.stripEmptyTags.keepTags = tr,td
			\end{lstlisting}

	\end{itemize}

	\greektext\underline{Σημαντικό:} εάν χρησιμοποιείται αυτή η ρύθμιση, η διαμόρφωση \latintext \texttt{stripEmptyTags.tags}
		\greektext δεν έχει πλέον καμία επίδραση. Μπορείτε να χρησιμοποιήσετε μία επιλογή κάθε φορά.

\end{frame}
