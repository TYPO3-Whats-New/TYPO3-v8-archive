\begin{frame}[fragile]
	\frametitle{Funciones Obsoletas/Eliminadas}
	\framesubtitle{Miscelánea}

	\begin{itemize}

		\item Las siguientes opciones de configuración han sido eliminadas:

			\begin{itemize}
				\item \texttt{\$TYPO3\_CONF\_VARS['SYS']['t3lib\_cs\_utils']}
				\item \texttt{\$TYPO3\_CONF\_VARS['SYS']['t3lib\_cs\_convMethod']}
			\end{itemize}

			\small
				(la funcionalidad es ahora auto-detectada y se usa \texttt{mbstring}
				por defecto si está disponible)
			\normalsize

		\item La propiedad obsoleta TypoScript \texttt{page.includeJSlibs} ha
			sido eliminada. Use en su lugar la propiedad TypoScript \texttt{page.includeJSLibs}
			("L" mayúscula)

		\item La opción de TypoScript \texttt{config.renderCharset}, que era usada como conjunto de caracteres para conversión interna dentro de una petición frontend,
			ha sido eliminada

	\end{itemize}

\end{frame}
